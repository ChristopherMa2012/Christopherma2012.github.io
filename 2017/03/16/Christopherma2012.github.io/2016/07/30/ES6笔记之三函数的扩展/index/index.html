ES6笔记之三函数的扩展

函数参数的默认值

1、ES6允许为函数的参数设置默认值，即直接写在参数定义的后面 

function a(x = 3,y = 4){
    console.log(x);
    console.log(y);
}
a(); //3 , 4
a(2,4)  // 2 , 4

2、结合解构赋值设置默认值 

function foo({x,y  = 3}){  //参数是一个对象，只有传入的参数也是一个对象时，才会进行解构
    console.log(x);
    console.log(y);
}
foo({})                  // x = undefined , y = 3
foo({x = 4 , y = 1})     //x = 4 , y = 1 
-----------------------------------------------------------------------------------

function bar([x = 2,y] = [ 3, 4]){  //[3 , 4] 是为参数变量[ x = 2, y]设置的默认值
    console.log(x);
    console.log(y);
}
bar();        //x = 3 , y = 4
bar([3,45])   //x = 3 , y = 45
-----------------------------------------------------------------------------------

function bar([x = 2,y]){  //参数变量[ x = 2, y]没有设置默认值
    console.log(x);
    console.log(y);
}
bar();        //报错：VM425:2 Uncaught TypeError: Cannot read property 
              // 'Symbol(Symbol.iterator)' of undefined(…)
bar([3,45])   //x = 3 , y = 45
